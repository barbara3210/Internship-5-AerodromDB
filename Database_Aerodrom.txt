-- Database: Aerodrom

-- DROP DATABASE IF EXISTS "Aerodrom";

CREATE DATABASE "Aerodrom"
    WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'English_United States.1252'
    LC_CTYPE = 'English_United States.1252'
    LOCALE_PROVIDER = 'libc'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1
    IS_TEMPLATE = False;
	
CREATE TABLE Aerodrom (
    id_aerodrom SERIAL PRIMARY KEY,
    naziv VARCHAR(100) NOT NULL,
    grad VARCHAR(30)
);

CREATE TABLE Avion (
    id_avion SERIAL PRIMARY KEY,
    naziv VARCHAR(100) NOT NULL,
	kapacitet_piste INT,
    kapacitet_skladista INT,
	status VARCHAR(50),
	datum_proizvodnje DATE
);

CREATE TABLE Let (
    id_let SERIAL PRIMARY KEY,
    datum DATE,
    broj_prodanih_karata INT,
    ocjena INT,
    aerodrom_polaska_id INT REFERENCES Aerodrom(id_aerodrom),
    aerodrom_dolaska_id INT REFERENCES Aerodrom(id_aerodrom),
    avion_id INT REFERENCES Avion(id_avion)
);

ALTER TABLE Let ADD CONSTRAINT check_datum CHECK (datum >= CURRENT_DATE);



CREATE TABLE Let_Osoblje(
	id_let INT REFERENCES Let(id_let),
	id_osoblje INT REFERENCES Osoblje(id_osoblje),
	id_pilot INT REFERENCES Pilot(id_pilot)
);

ALTER TABLE Let_Osoblje ADD FOREIGN KEY (id_let_osoblje) REFERENCES Let(id_let);



CREATE TABLE Karta (
    id_karta SERIAL PRIMARY KEY,
    serijski_broj VARCHAR(100) NOT NULL,
	putnik_id INT REFERENCES Putnik(id_putnik),
	let_id INT REFERENCES Let(id_let),
	sjedalo VARCHAR(255),
	cijena DECIMAL
);

CREATE TABLE Putnik (
    id_putnik SERIAL PRIMARY KEY,
    ime VARCHAR(30) NOT NULL,
	prezime VARCHAR(30) NOT NULL,
	datum_rodenja DATE,
	prebivaliste VARCHAR(50) NOT NULL,
	email VARCHAR(50)
	
);

CREATE TABLE Pilot (
    id_pilot SERIAL PRIMARY KEY,
    ime VARCHAR(30) NOT NULL,
	prezime VARCHAR(30) NOT NULL,
	datum_rodenja DATE,
	placa DECIMAL	
);

CREATE TABLE Osoblje (
    id_osoblje SERIAL PRIMARY KEY,
    ime VARCHAR(30) NOT NULL,
	prezime VARCHAR(30) NOT NULL,
	datum_rodenja DATE
	
);

CREATE TABLE Loyalty_Kartica (
    id_loyalty SERIAL PRIMARY KEY,
    korisnik_id INT REFERENCES Putnik(id_putnik) UNIQUE,
    trajanje_godina INT,
    broj_kupljenih_karata INT
);

--●	ispis imena i modela svih aviona s kapacitetom većim od 100
SELECT naziv FROM Avion 
WHERE kapacitet_piste + kapacitet_skladista > 100;

--●	ispis svih karata čija je cijena između 100 i 200 eura
SELECT * FROM Karta 
WHERE cijena BETWEEN 100 AND 200;

--●	ispis svih pilotkinja s više od 20 odrađenih letova do danas
SELECT * FROM Pilot 
WHERE id_pilot 
IN (SELECT id_pilot FROM Let_Osoblje 
GROUP BY id_pilot HAVING COUNT(id_pilot) > 20);

--●	ispis svih domaćina/ca zrakoplova koji su trenutno u zraku
SELECT  Osoblje.ime, Osoblje.prezime
FROM Osoblje
JOIN Let_Osoblje ON Osoblje.id_osoblje = Let_Osoblje.id_osoblje
JOIN Let ON Let_Osoblje.id_let = Let.id_let
WHERE Let.datum BETWEEN NOW() - INTERVAL 30 MINUTE AND NOW();

--●	ispis broja letova u Split/iz Splita 2023. godine
SELECT COUNT(*) AS broj_letova
FROM Let
WHERE YEAR(datum) = 2023 
AND (aerodrom_polaska_id =(SELECT id_aerodrom FROM Aerodrom WHERE grad = 'Split') 
	 OR aerodrom_dolaska_id = (SELECT id_aerodrom FROM Aerodrom WHERE grad = 'Split'));

--●	ispis svih letova za Beč u prosincu 2023.
SELECT * FROM Let
WHERE YEAR(datum) = 2023 AND MONTH(datum) = 12 
AND (aerodrom_polaska_id = (SELECT id_aerodrom FROM Aerodrom WHERE grad = 'Beč') 
	 OR aerodrom_dolaska_id = (SELECT id_aerodrom FROM Aerodrom WHERE grad = 'Beč'));

--●	ispis prosječne ocjene letova kompanije AirDUMP
SELECT AVG(ocjena) AS prosjecna_ocjena
FROM Let
WHERE avion_id IN (SELECT id_avion FROM Avion WHERE naziv = 'AirDUMP');

--●	ispis svih aerodroma u Londonu, sortiranih po broju Airbus aviona trenutno na njihovim pistama
SELECT Aerodrom.naziv FROM Aerodrom
WHERE Aerodrom.grad = 'London';

--●	smanjite cijenu za 20% svim kartama čiji letovi imaju manje od 20 ljudi
UPDATE Karta
SET cijena = cijena * 0.8
WHERE let_id IN (SELECT id_let FROM Let WHERE broj_prodanih_karata < 20);


--●	povisite plaću za 100 eura svim pilotima koji su ove godine imali više od 10 letova duljih od 10 sati
UPDATE Pilot
SET placa = placa + 100
WHERE id_pilot IN (
    SELECT id_pilot
    FROM Let_Osoblje
    JOIN Let ON Let_Osoblje.id_let = Let.id_let
    WHERE EXTRACT(YEAR FROM Let.datum) = EXTRACT(YEAR FROM CURRENT_DATE) AND Let.trajanje > '10:00:00'
    GROUP BY id_pilot
    HAVING COUNT(Let_Osoblje.id_pilot) > 10
);

--●	razmontirajte avione starije od 20 godina koji nemaju letove pred sobom
UPDATE Avion
SET status = 'razmontiran'
WHERE EXTRACT(YEAR FROM AGE(NOW(), datum_proizvodnje)) > 20
  AND id_avion NOT IN (SELECT DISTINCT avion_id FROM Let);

--●	izbrišite sve letove koji nemaju ni jednu prodanu kartu
DELETE FROM Let
WHERE id_let NOT IN (SELECT DISTINCT let_id FROM Karta);

--●	izbrišite sve kartice vjernosti putnika čije prezime završava na -ov/a, -in/a
DELETE FROM Loyalty_Kartica
WHERE korisnik_id IN (SELECT id_putnik FROM Putnik WHERE prezime ILIKE '%ov' OR prezime ILIKE '%in');
